126void compute(size_t n, const Float_* mean, const Float_* variance, Float_* fitted, Float_* residuals, Workspace<Float_>& workspace, const Options& options) {
127 auto& xbuffer = workspace.xbuffer;
128 xbuffer.resize(n);
129 auto& ybuffer = workspace.ybuffer;
130 ybuffer.resize(n);
131
132 auto quad = [](Float_ x) -> Float_ {
133 return x * x * x * x;
134 };
135
136 size_t counter = 0;
137 Float_ min_mean = options.minimum_mean;
138 for (size_t i = 0; i < n; ++i) {
139 if (!options.mean_filter || mean[i] >= min_mean) {
140 xbuffer[counter] = mean[i];
141 if (options.transform) {
142 ybuffer[counter] = std::pow(variance[i], 0.25); // Using the same quarter-root transform that limma::voom uses.
143 } else {
144 ybuffer[counter] = variance[i];
145 }
146 ++counter;
147 }
148 }
149
150 if (counter < 2) {
151 throw std::runtime_error("not enough observations above the minimum mean");
152 }
153
154 auto& sorter = workspace.sorter;
155 sorter.set(counter, xbuffer.data());
156 auto& work = workspace.sort_workspace;
157 sorter.permute(std::array<Float_*, 2>{ xbuffer.data(), ybuffer.data() }, work);
158
159 WeightedLowess::Options<Float_> smooth_opt;
160 if (options.use_minimum_width) {
161 smooth_opt.span = options.minimum_window_count;
162 smooth_opt.span_as_proportion = false;
163 smooth_opt.minimum_width = options.minimum_width;
164 } else {
165 smooth_opt.span = options.span;
166 }
167 smooth_opt.num_threads = options.num_threads;
168
169 // Using the residual array to store the robustness weights as a placeholder;
170 // we'll be overwriting this later.
171 WeightedLowess::compute(counter, xbuffer.data(), ybuffer.data(), fitted, residuals, smooth_opt);
172
173 // Determining the left edge before we unpermute.
174 Float_ left_x = xbuffer[0];
175 Float_ left_fitted = (options.transform ? quad(fitted[0]) : fitted[0]);
176
177 sorter.unpermute(fitted, work);
178
179 // Walking backwards to shift the elements back to their original position
180 // (i.e., before filtering on the mean) on the same array. We need to walk
181 // backwards to ensure that writing to the original position on this array
182 // doesn't clobber the first 'counter' positions containing the fitted
183 // values, at least not until each value is shifted to its original place.
184 for (size_t i = n; i > 0; --i) {
185 auto j = i - 1;
186 if (!options.mean_filter || mean[j] >= min_mean) {
187 --counter;
188 fitted[j] = (options.transform ? quad(fitted[counter]) : fitted[counter]);
189 } else {
190 fitted[j] = mean[j] / left_x * left_fitted; // draw a y = x line to the origin from the left of the fitted trend.
191 }
192 }
193
194 for (size_t i = 0; i < n; ++i) {
195 residuals[i] = variance[i] - fitted[i];
196 }
197 return;
198}
