76 auto NR = mat->nrow(), NC = mat->ncol();

78 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
79 std::vector<Stat_> tmp_means(blocked ? nblocks : 0);
80 std::vector<Stat_> tmp_vars(blocked ? nblocks : 0);

82 std::vector<Value_> buffer(NC);

284template<typename Value_, typename Index_, typename Block_, typename Stat_>
285void compute_blocked(
286 const tatami::Matrix<Value_, Index_>* mat,
287 const Block_* block,
288 std::vector<Stat_*> means,
289 std::vector<Stat_*> variances,
290 std::vector<Stat_*> fitted,
291 std::vector<Stat_*> residuals,
292 Stat_* ave_means,
293 Stat_* ave_variances,
294 Stat_* ave_fitted,
295 Stat_* ave_residuals,
296 const Options& options)
297{
298 Index_ NR = mat->nrow(), NC = mat->ncol();
299 std::vector<Index_> block_size;
300
301 if (block) {
302 block_size = tatami_stats::tabulate_groups(block, NC);
303 internal::compute(mat, means, variances, block, block_size, options.num_threads);
304 } else {
305 block_size.push_back(NC); // everything is one big block.
306 internal::compute(mat, means, variances, block, block_size, options.num_threads);
307 }
308
309 fit_variance_trend::Workspace<Stat_> work;
310 auto fopt = options.fit_variance_trend_options;
311 fopt.num_threads = options.num_threads;
312 for (size_t b = 0, nblocks = block_size.size(); b < nblocks; ++b) {
313 if (block_size[b] >= 2) {
314 fit_variance_trend::compute(NR, means[b], variances[b], fitted[b], residuals[b], work, fopt);
315 } else {
316 std::fill(fitted[b], fitted[b] + NR, std::numeric_limits<double>::quiet_NaN());
317 std::fill(residuals[b], residuals[b] + NR, std::numeric_limits<double>::quiet_NaN());
318 }
319 }
320
321 if (ave_means || ave_variances || ave_fitted || ave_residuals) {
322 std::vector<double> block_weight = block_weights::compute(block_size, options.block_weight_policy, options.variable_block_weight_parameters);
323
324 // Check whether there might be NaNs for very small blocks,
325 // in which case we need to ignore them.
326 Index_ min_block = NC;
327 for (size_t b = 0, nblocks = block_size.size(); b < nblocks; ++b) {
328 if (block_size[b] < min_block) {
329 min_block = block_size[b];
330 }
331 }
332
333 if (ave_means) {
334 average_vectors::compute_weighted(NR, means, block_weight.data(), ave_means, min_block < 1);
335 }
336 if (ave_variances) {
337 average_vectors::compute_weighted(NR, variances, block_weight.data(), ave_variances, min_block < 2);
338 }
339 if (ave_fitted) {
340 average_vectors::compute_weighted(NR, fitted, block_weight.data(), ave_fitted, min_block < 2);
341 }
342 if (ave_residuals) {
343 average_vectors::compute_weighted(NR, residuals, block_weight.data(), ave_residuals, min_block < 2);
344 }
345 }
